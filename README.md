```
while True:
    i+=1
    a_y[i] = -g - (k*v_y[i-1])/m
    a_x[i] = -(k*v_x[i-1])/m

    v_y[i] = v_y[i-1] + a_y[i]*dt
    v_x[i] = v_x[i-1] + a_x[i]*dt
```
